## 概述
本文档基于`/docs/base/`目录中的需求文档、数据模型设计、数据库配置和接口设计规范，详细说明预约系统后端的测试策略、测试配置和测试用例设计。

## 测试技术栈

### 核心测试框架
- **Jest 29+**: 主要测试框架，提供测试运行、断言和覆盖率报告
- **TestContainers**: 用于集成测试，提供真实的PostgreSQL数据库容器
- **Supertest**: HTTP请求测试库，用于API端点测试
- **@nestjs/testing**: NestJS官方测试工具包

### 测试类型
- **单元测试**: 测试单个函数、类或方法的逻辑
- **集成测试**: 测试模块间集成和数据库操作
- **端到端测试**: 测试完整的API请求响应流程
- **性能测试**: 测试系统在高负载下的表现

### TestContainers配置 (test/setup.ts)
```typescript
import { Test } from '@nestjs/testing';
import { PostgreSqlContainer } from '@testcontainers/postgresql';
import { PrismaService } from '../src/modules/prisma/prisma.service';

let postgresContainer: PostgreSqlContainer;

// 全局测试设置
beforeAll(async () => {
  // 启动PostgreSQL容器
  postgresContainer = await new PostgreSqlContainer()
    .withDatabase('test_booking_system')
    .withUsername('postgres')
    .withPassword('test_password')
    .start();

  // 设置环境变量
  process.env.DATABASE_URL = postgresContainer.getConnectionUri();

  // 创建测试模块
  const moduleRef = await Test.createTestingModule({
    imports: [AppModule],
  }).compile();

  const app = moduleRef.createNestApplication();
  await app.init();

  // 运行数据库迁移
  const prismaService = app.get(PrismaService);
  await prismaService.$executeRaw`CREATE EXTENSION IF NOT EXISTS "uuid-ossp";`;
  await prismaService.$executeRaw`CREATE EXTENSION IF NOT EXISTS "pgcrypto";`;
});

// 全局测试清理
afterAll(async () => {
  await postgresContainer.stop();
});
```

```typescript
describe('UsersService Integration', () => {
  let service: UsersService;
  let prisma: PrismaService;
  let postgresContainer: PostgreSqlContainer;

  beforeAll(async () => {
    postgresContainer = await new PostgreSqlContainer()
      .withDatabase('test_users')
      .start();

    process.env.DATABASE_URL = postgresContainer.getConnectionUri();

    const moduleRef = await Test.createTestingModule({
      imports: [AppModule],
    }).compile();

    const app = moduleRef.createNestApplication();
    await app.init();

    service = app.get(UsersService);

    // 运行数据库迁移

  });

  afterAll(async () => {
    await prisma.$disconnect();
    await postgresContainer.stop();
  });

  beforeEach(async () => {
    await prisma.user.deleteMany();
  });

  it('应该在真实数据库中创建用户', async () => {
    const createUserDto: CreateUserDto = {
      name: '集成测试用户',
      phone: '13800138001',
    };

    expect(user).toHaveProperty('id');
    expect(user.name).toBe('集成测试用户');

    expect(user.phoneHash).toBeDefined();

    // 验证用户已保存到数据库
    const dbUser = await prisma.user.findUnique({
      where: { id: user.id },
    });
    expect(dbUser).toBeDefined();
    expect(dbUser?.name).toBe('集成测试用户');
  });
});
```

```typescript
describe('UsersController (e2e)', () => {
  let app: INestApplication;
  let prisma: PrismaService;

  beforeAll(async () => {
    const moduleFixture: TestingModule = await Test.createTestingModule({
      imports: [AppModule],
    }).compile();

    app = moduleFixture.createNestApplication();
    await app.init();

    prisma = app.get(PrismaService);
  });

  afterAll(async () => {
    await app.close();
  });

  describe('GET /users', () => {
    it('应该返回用户列表', async () => {
      // 创建测试数据
      await prisma.user.create({
        data: {
          name: '测试用户1',
          phone: '13800138001',
          phoneHash: 'hash-13800138001',
        },
      });

      const response = await request(app.getHttpServer())
        .get('/users')
        .expect(200);

      expect(response.body).toHaveProperty('data');
      expect(response.body.data).toBeInstanceOf(Array);
      expect(response.body.data[0].name).toBe('测试用户1');
    });
  });

  describe('POST /users', () => {
    it('应该创建新用户', async () => {
      const createUserDto = {
        name: '新用户',
        phone: '13800138002',
        email: 'new@example.com',
      };

      const response = await request(app.getHttpServer())
        .post('/users')
        .send(createUserDto)
        .expect(201);

      expect(response.body).toHaveProperty('id');
      expect(response.body.name).toBe('新用户');
      expect(response.body.phone).toBe('138****8002');

      // 验证用户已保存
      const user = await prisma.user.findUnique({
        where: { phoneHash: expect.any(String) },
      });
      expect(user).toBeDefined();
    });

    it('手机号重复时应返回错误', async () => {
      // 先创建用户
      await prisma.user.create({
        data: {
          name: '已存在用户',
          phone: '13800138003',
          phoneHash: 'hash-13800138003',
        },
      });

      const createUserDto = {
        name: '重复用户',
        phone: '13800138003',
      };

      const response = await request(app.getHttpServer())
        .post('/users')
        .send(createUserDto)
        .expect(409);

      expect(response.body).toHaveProperty('message');
      expect(response.body.message).toContain('手机号 13800138003 已存在');
    });
  });
});
```
